-- [[ VANDER GUARD ENFORCER ]] --
local VG_KEY = "PASTE_KEY_HERE"
local function authenticate()
    local hwid = game:GetService("RbxAnalyticsService"):GetClientId()
    local url = "https://vander.guard/v2/auth?key="..VG_KEY.."&hwid="..hwid
    local req = game:HttpGet(url)
    local res = game:GetService("HttpService"):JSONDecode(req)
    
    if res.Success then
        print("[VG] ACCESS GRANTED")
        loadstring(res.ScriptData)()
    else
        game:GetService("Players").LocalPlayer:Kick("[VG] ERROR: " .. res.Error)
    end
end
authenticate()
